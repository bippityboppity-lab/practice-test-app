<script>
// EMBEDDED QUESTION BANK
// To use external JSON: Comment out quizData below and uncomment loadQuestions() at bottom
let quizData = [];

let currentQuestion = 0;
let userAnswers = {};
let questionOrder = [];
let showAnswers = false;
let showExplanations = false;
let markedQuestions = new Set();
let reviewMode = null;

// Load questions from external JSON file (DISABLED - using embedded data)
async function loadQuestions() {
try {
const response = await fetch('questions.json');
if (!response.ok) {
throw new Error('Failed to load questions');
}
quizData = await response.json();
loadQuestions();
} catch (error) {
console.error('Error loading questions:', error);
document.getElementById('loadingSection').classList.add('hidden');
document.getElementById('errorSection').classList.remove('hidden');
}
}

function init() {
questionOrder = [...Array(quizData.length).keys()];
document.getElementById('loadingSection').classList.add('hidden');
document.getElementById('quizSection').classList.remove('hidden');
document.getElementById('questionCount').textContent = `${quizData.length} Questions`;
createNavigation();
loadQuestion();
}

function createNavigation() {
const navGrid = document.getElementById('navGrid');
navGrid.innerHTML = '';

questionOrder.forEach((qIdx, position) => {
const btn = document.createElement('button');
btn.className = 'nav-btn';
btn.textContent = quizData[qIdx].number;
btn.onclick = () => goToQuestion(position);

if (position === currentQuestion) {
btn.classList.add('current');
}
if (userAnswers[qIdx]?.checked) {
btn.classList.add('answered');
}
if (markedQuestions.has(qIdx)) {
btn.classList.add('marked');
}

navGrid.appendChild(btn);
});
}

function loadQuestion() {
const qIdx = questionOrder[currentQuestion];
const question = quizData[qIdx];

document.getElementById('questionNumber').textContent =
`Question ${currentQuestion + 1} of ${questionOrder.length} (Q${question.number})`;
document.getElementById('questionText').textContent = question.question;

// Update mark button
const markBtn = document.getElementById('markBtn');
if (markedQuestions.has(qIdx)) {
markBtn.classList.add('marked');
markBtn.textContent = '⭐ Marked';
} else {
markBtn.classList.remove('marked');
markBtn.textContent = '⭐ Mark';
}

// Load options
const container = document.getElementById('optionsContainer');
container.innerHTML = '';

Object.entries(question.options).forEach(([letter, text]) => {
const div = document.createElement('div');
div.className = 'option';

const checkbox = document.createElement('input');
checkbox.type = 'checkbox';
checkbox.id = `opt_${letter}`;
checkbox.checked = userAnswers[qIdx]?.selected?.includes(letter) || false;
checkbox.onchange = () => selectOption(letter);

const label = document.createElement('label');
label.className = 'option-text';
label.htmlFor = `opt_${letter}`;
label.innerHTML = `<strong>${letter}.</strong> ${text}`;

div.appendChild(checkbox);
div.appendChild(label);
div.onclick = (e) => {
if (e.target !== checkbox) {
checkbox.checked = !checkbox.checked;
selectOption(letter);
}
};

container.appendChild(div);
});

// Update UI state
updateQuestionUI();
updateProgress();
createNavigation();
}

function selectOption(letter) {
const qIdx = questionOrder[currentQuestion];
if (!userAnswers[qIdx]) {
userAnswers[qIdx] = { selected: [], checked: false };
}

const selected = userAnswers[qIdx].selected;
const index = selected.indexOf(letter);

if (index > -1) {
selected.splice(index, 1);
} else {
selected.push(letter);
}

userAnswers[qIdx].checked = false;
updateQuestionUI();
}

function checkAnswer() {
const qIdx = questionOrder[currentQuestion];
const question = quizData[qIdx];
const userAnswer = userAnswers[qIdx];

if (!userAnswer || userAnswer.selected.length === 0) {
alert('Please select at least one answer!');
return;
}

const correct = arraysEqual(
userAnswer.selected.sort(),
question.answer.sort()
);

userAnswer.checked = true;
userAnswer.correct = correct;

// Show feedback
const feedback = document.getElementById('feedback');
const feedbackTitle = document.getElementById('feedbackTitle');
const feedbackText = document.getElementById('feedbackText');

feedback.classList.remove('hidden', 'correct', 'incorrect');

if (correct) {
feedback.classList.add('correct');
feedbackTitle.textContent = '✓ Correct!';
feedbackText.textContent = `The answer is: ${question.answer.join(', ')}`;
} else {
feedback.classList.add('incorrect');
feedbackTitle.textContent = '✗ Incorrect';
feedbackText.textContent = `The correct answer is: ${question.answer.join(', ')}. You selected: ${userAnswer.selected.join(', ')}`;
}

// Show explanation
const explanationContainer = document.getElementById('explanationText');
explanationContainer.innerHTML = '';

if (Array.isArray(question.explanation)) {
question.explanation.forEach(expl => {
const explDiv = document.createElement('div');
explDiv.className = 'explanation-item';

if (expl.Status === 'Correct') {
explDiv.classList.add('correct-explanation');
} else if (expl.Status === 'Incorrect') {
explDiv.classList.add('incorrect-explanation');
}

const statusDiv = document.createElement('div');
statusDiv.className = 'explanation-status';
statusDiv.classList.add(expl.Status.toLowerCase());
statusDiv.textContent = expl.Status;

const choiceDiv = document.createElement('div');
choiceDiv.className = 'explanation-choice';
choiceDiv.textContent = expl.Choice;

const rationaleDiv = document.createElement('div');
rationaleDiv.className = 'explanation-rationale';
rationaleDiv.textContent = expl.Rationale;

explDiv.appendChild(statusDiv);
explDiv.appendChild(choiceDiv);
explDiv.appendChild(rationaleDiv);
explanationContainer.appendChild(explDiv);
});
} else {
// Fallback for string explanations
const p = document.createElement('p');
p.textContent = question.explanation;
explanationContainer.appendChild(p);
}

document.getElementById('explanationBox').classList.remove('hidden');

// Highlight options
Object.keys(question.options).forEach(letter => {
const checkbox = document.getElementById(`opt_${letter}`);
const div = checkbox.parentElement;

if (question.answer.includes(letter)) {
div.classList.add('correct');
} else if (userAnswer.selected.includes(letter)) {
div.classList.add('incorrect');
}
});

createNavigation();
}

function updateQuestionUI() {
const qIdx = questionOrder[currentQuestion];
const userAnswer = userAnswers[qIdx];

const feedback = document.getElementById('feedback');
const explanation = document.getElementById('explanationBox');
const submitBtn = document.getElementById('submitBtn');

if (userAnswer?.checked) {
submitBtn.textContent = 'Checked ✓';
submitBtn.disabled = true;
submitBtn.style.opacity = '0.6';

if (!showAnswers) {
feedback.classList.add('hidden');
}
if (!showExplanations) {
explanation.classList.add('hidden');
}
} else {
submitBtn.textContent = 'Check Answer';
submitBtn.disabled = false;
submitBtn.style.opacity = '1';
feedback.classList.add('hidden');
explanation.classList.add('hidden');

// Remove highlighting
document.querySelectorAll('.option').forEach(opt => {
opt.classList.remove('correct', 'incorrect');
});
}
}

function resetQuestionUI() {
document.getElementById('feedback').classList.add('hidden');
document.getElementById('explanationBox').classList.add('hidden');
document.querySelectorAll('.option').forEach(opt => {
opt.classList.remove('correct', 'incorrect');
});
}

function updateProgress() {
const answered = Object.values(userAnswers).filter(a => a.checked).length;
const percentage = (answered / quizData.length) * 100;
document.getElementById('progressBar').style.width = percentage + '%';
}

function toggleMark() {
const qIdx = questionOrder[currentQuestion];
if (markedQuestions.has(qIdx)) {
markedQuestions.delete(qIdx);
} else {
markedQuestions.add(qIdx);
}
loadQuestion();
}

function toggleAnswers() {
showAnswers = !showAnswers;
const btn = event.target;
btn.classList.toggle('active');

if (showAnswers) {
document.getElementById('feedback').classList.remove('hidden');
} else {
document.getElementById('feedback').classList.add('hidden');
}
}

function toggleExplanations() {
showExplanations = !showExplanations;
const btn = event.target;
btn.classList.toggle('active');

if (showExplanations) {
document.getElementById('explanationBox').classList.remove('hidden');
} else {
document.getElementById('explanationBox').classList.add('hidden');
}
}

function shuffleQuestions() {
if (confirm('Shuffle questions? This will reset your progress.')) {
questionOrder = [...Array(quizData.length).keys()];
for (let i = questionOrder.length - 1; i > 0; i--) {
const j = Math.floor(Math.random() * (i + 1));
[questionOrder[i], questionOrder[j]] = [questionOrder[j], questionOrder[i]];
}
currentQuestion = 0;
userAnswers = {};
markedQuestions.clear();
resetQuestionUI();
loadQuestion();
}
}

function nextQuestion() {
if (currentQuestion < questionOrder.length - 1) {
currentQuestion++;
resetQuestionUI();
loadQuestion();
} else if (reviewMode) {
currentQuestion = 0;
resetQuestionUI();
loadQuestion();
}
}

function previousQuestion() {
if (currentQuestion > 0) {
currentQuestion--;
resetQuestionUI();
loadQuestion();
} else if (reviewMode) {
currentQuestion = questionOrder.length - 1;
resetQuestionUI();
loadQuestion();
}
}

function goToQuestion(position) {
currentQuestion = position;
resetQuestionUI();
loadQuestion();
}

function arraysEqual(arr1, arr2) {
if (arr1.length !== arr2.length) return false;
return arr1.every((val, idx) => val === arr2[idx]);
}

function resetQuiz() {
if (reviewMode || confirm('Reset quiz? This will clear all your answers.')) {
currentQuestion = 0;
userAnswers = {};
markedQuestions.clear();
reviewMode = null;
questionOrder = [...Array(quizData.length).keys()];
document.getElementById('reviewBanner').classList.add('hidden');
document.getElementById('resultsSection').classList.add('hidden');
document.getElementById('quizSection').classList.remove('hidden');
createNavigation();
resetQuestionUI();
loadQuestion();
}
}

function toggleReviewMode(mode) {
reviewMode = mode;
let filteredQuestions = [];

if (mode === 'all') {
filteredQuestions = [...Array(quizData.length).keys()];
showReviewBanner('Reviewing All Questions');
} else if (mode === 'marked') {
filteredQuestions = Array.from(markedQuestions);
if (filteredQuestions.length === 0) {
alert('No questions marked for review!');
return;
}
showReviewBanner(`Reviewing ${filteredQuestions.length} Marked Questions`);
} else if (mode === 'incorrect') {
filteredQuestions = Object.keys(userAnswers)
.filter(idx => userAnswers[idx].checked && !userAnswers[idx].correct)
.map(idx => parseInt(idx));
if (filteredQuestions.length === 0) {
alert('No incorrect answers to review!');
return;
}
showReviewBanner(`Reviewing ${filteredQuestions.length} Incorrect Questions`);
}

questionOrder = filteredQuestions;
currentQuestion = 0;
document.getElementById('resultsSection').classList.add('hidden');
document.getElementById('quizSection').classList.remove('hidden');
createNavigation();
resetQuestionUI();
loadQuestion();
}

function showReviewBanner(text) {
const banner = document.getElementById('reviewBanner');
banner.textContent = text + ' - Click Reset Quiz to exit review mode';
banner.classList.remove('hidden');
}

function showResults() {
const answered = Object.values(userAnswers).filter(a => a.checked).length;
const correct = Object.values(userAnswers).filter(a => a.checked && a.correct).length;
const incorrect = answered - correct;
const unanswered = quizData.length - answered;
const percentage = Math.round((correct / quizData.length) * 100);

document.getElementById('scoreNumber').textContent = percentage + '%';
document.getElementById('correctCount').textContent = correct;
document.getElementById('incorrectCount').textContent = incorrect;
document.getElementById('unansweredCount').textContent = unanswered;

document.getElementById('quizSection').classList.add('hidden');
document.getElementById('resultsSection').classList.remove('hidden');
}

function reviewAll() {
toggleReviewMode('all');
}

function reviewIncorrect() {
toggleReviewMode('incorrect');
}

// Start with embedded questions (for offline use)
// To use external JSON, uncomment: loadQuestions();
loadQuestions();
</script>
